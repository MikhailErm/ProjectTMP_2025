120// Основная функция, реализующая алгоритм SHA-256
121inline QString sha256(QString input) {
122 // Получаем длину исходной строки
123 int size = input.length();
124
125 // Подготовка сообщения: дополнение до кратности 512 бит (64 байта)
126 QByteArray message(size + 1 + (64 - (size + 1) % 64), 0);
127
128 // Копируем входные данные
129 for (int i = 0; i < size; i++) {
130 message[i] = input[i].toLatin1();
131 }
132
133 // Добавляем бит '1' после данных (в шестнадцатеричном — 0x80)
134 message[size] = 0x80;
135
136 // Дописываем длину сообщения (в битах) в конец блока (64 бита)
137 quint64 bitLength = size * 8;
138 for (int i = 0; i < 8; i++) {
139 message[message.size() - 8 + i] = (bitLength >> (56 - i * 8)) & 0xFF;
140 }
141
142 // Инициализация переменных хэша
143 uint32_t h0 = 0x6a09e667;
144 uint32_t h1 = 0xbb67ae85;
145 uint32_t h2 = 0x3c6ef372;
146 uint32_t h3 = 0xa54ff53a;
147 uint32_t h4 = 0x510e527f;
148 uint32_t h5 = 0x9b05688c;
149 uint32_t h6 = 0x1f83d9ab;
150 uint32_t h7 = 0x5be0cd19;
151
152 // Обработка каждого 512-битного блока
153 for (int i = 0; i < message.size(); i += 64) {
154 uint32_t words[64];
155
156 // Разбиение блока на 16 32-битных слов (big-endian)
157 for (int j = 0; j < 16; j++) {
158 words[j] = (static_cast<uint8_t>(message[i + j * 4]) << 24)
159 | (static_cast<uint8_t>(message[i + j * 4 + 1]) << 16)
160 | (static_cast<uint8_t>(message[i + j * 4 + 2]) << 8)
161 | static_cast<uint8_t>(message[i + j * 4 + 3]);
162 }
163
164 // Расширение до 64 слов
165 for (int j = 16; j < 64; j++) {
166 words[j] = smallSigma1(words[j - 2]) + words[j - 7] + smallSigma0(words[j - 15]) + words[j - 16];
167 }
168
169 // Инициализация рабочих переменных
170 uint32_t a = h0, b = h1, c = h2, d = h3;
171 uint32_t e = h4, f = h5, g = h6, h = h7;
172
173 // Основной цикл SHA-256 (64 раунда)
174 for (int j = 0; j < 64; j++) {
175 uint32_t temp1 = h + bigSigma1(e) + choose(e, f, g) + constants[j] + words[j];
176 uint32_t temp2 = bigSigma0(a) + majority(a, b, c);
177
178 h = g;
179 g = f;
180 f = e;
181 e = d + temp1;
182 d = c;
183 c = b;
184 b = a;
185 a = temp1 + temp2;
186 }
187
188 // Прибавляем результат к текущим значениям хэша
189 h0 += a; h1 += b; h2 += c; h3 += d;
190 h4 += e; h5 += f; h6 += g; h7 += h;
191 }
192
193 // Возвращаем хэш как строку в шестнадцатеричном формате
194 return QString("%1%2%3%4%5%6%7%8")
195 .arg(h0, 8, 16, QLatin1Char('0'))
196 .arg(h1, 8, 16, QLatin1Char('0'))
197 .arg(h2, 8, 16, QLatin1Char('0'))
198 .arg(h3, 8, 16, QLatin1Char('0'))
199 .arg(h4, 8, 16, QLatin1Char('0'))
200 .arg(h5, 8, 16, QLatin1Char('0'))
201 .arg(h6, 8, 16, QLatin1Char('0'))
202 .arg(h7, 8, 16, QLatin1Char('0'))
203 .toUpper();
204}

345 static uint64_t modExp(uint64_t base, uint64_t exp, uint64_t mod) {
346 uint64_t result = 1;
347 base = base % mod;
348 while (exp > 0) {
349 if (exp & 1) result = (result * base) % mod;
350 exp >>= 1;
351 base = (base * base) % mod;
352 }

377 static uint64_t modInverse(uint64_t a, uint64_t m) {
378 int64_t m0 = m, t, q;
379 int64_t x0 = 0, x1 = 1;
380 if (m == 1) return 0;
381 while (a > 1) {
382 q = a / m;
383 t = m;
384 m = a % m;
385 a = t;
386 t = x0;
387 x0 = x1 - q * x0;
388 x1 = t;
389 }
390 if (x1 < 0) x1 += m0;
391 return x1;
392 }
