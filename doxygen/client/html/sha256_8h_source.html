136// Основная функция, реализующая алгоритм SHA-256
137inline QString sha256(QString input) {
138 // Получаем длину исходной строки
139 int size = input.length();
140
141 // Подготовка сообщения: дополнение до кратности 512 бит (64 байта)
142 QByteArray message(size + 1 + (64 - (size + 1) % 64), 0);
143
144 // Копируем входные данные
145 for (int i = 0; i < size; i++) {
146 message[i] = input[i].toLatin1();
147 }
148
149 // Добавляем бит '1' после данных (в шестнадцатеричном — 0x80)
150 message[size] = 0x80;
151
152 // Дописываем длину сообщения (в битах) в конец блока (64 бита)
153 quint64 bitLength = size * 8;
154 for (int i = 0; i < 8; i++) {
155 message[message.size() - 8 + i] = (bitLength >> (56 - i * 8)) & 0xFF;
156 }
157
158 // Инициализация переменных хэша
159 uint32_t h0 = 0x6a09e667;
160 uint32_t h1 = 0xbb67ae85;
161 uint32_t h2 = 0x3c6ef372;
162 uint32_t h3 = 0xa54ff53a;
163 uint32_t h4 = 0x510e527f;
164 uint32_t h5 = 0x9b05688c;
165 uint32_t h6 = 0x1f83d9ab;
166 uint32_t h7 = 0x5be0cd19;
167
168 // Обработка каждого 512-битного блока
169 for (int i = 0; i < message.size(); i += 64) {
170 uint32_t words[64];
171
172 // Разбиение блока на 16 32-битных слов (big-endian)
173 for (int j = 0; j < 16; j++) {
174 words[j] = (static_cast<uint8_t>(message[i + j * 4]) << 24)
175 | (static_cast<uint8_t>(message[i + j * 4 + 1]) << 16)
176 | (static_cast<uint8_t>(message[i + j * 4 + 2]) << 8)
177 | static_cast<uint8_t>(message[i + j * 4 + 3]);
178 }
179
180 // Расширение до 64 слов
181 for (int j = 16; j < 64; j++) {
182 words[j] = smallSigma1(words[j - 2]) + words[j - 7] + smallSigma0(words[j - 15]) + words[j - 16];
183 }
184
185 // Инициализация рабочих переменных
186 uint32_t a = h0, b = h1, c = h2, d = h3;
187 uint32_t e = h4, f = h5, g = h6, h = h7;
188
189 // Основной цикл SHA-256 (64 раунда)
190 for (int j = 0; j < 64; j++) {
191 uint32_t temp1 = h + bigSigma1(e) + choose(e, f, g) + constants[j] + words[j];
192 uint32_t temp2 = bigSigma0(a) + majority(a, b, c);
193
194 h = g;
195 g = f;
196 f = e;
197 e = d + temp1;
198 d = c;
199 c = b;
200 b = a;
201 a = temp1 + temp2;
202 }
203
204 // Прибавляем результат к текущим значениям хэша
205 h0 += a; h1 += b; h2 += c; h3 += d;
206 h4 += e; h5 += f; h6 += g; h7 += h;
207 }
208
209 // Возвращаем хэш как строку в шестнадцатеричном формате
210 return QString("%1%2%3%4%5%6%7%8")
211 .arg(h0, 8, 16, QLatin1Char('0'))
212 .arg(h1, 8, 16, QLatin1Char('0'))
213 .arg(h2, 8, 16, QLatin1Char('0'))
214 .arg(h3, 8, 16, QLatin1Char('0'))
215 .arg(h4, 8, 16, QLatin1Char('0'))
216 .arg(h5, 8, 16, QLatin1Char('0'))
217 .arg(h6, 8, 16, QLatin1Char('0'))
218 .arg(h7, 8, 16, QLatin1Char('0'))
219 .toUpper();
220}
